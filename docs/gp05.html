include gdrive-sheets
include data-source
include shared-gdrive(
  "dcic-2021",
  "1wyQZj_L0qqV9Ekgr9au6RX2iqt2Ga8Ep")
import math as M
import statistics as S
import lists as L


ssid = "1RYN0i4Zx_UETVuYacgaGfnFcv4l9zd9toQTTdkQkj7g"
artists = 
  load-table: c1, c2, c3, c4, c5, c6, c7, c8
    source: load-spreadsheet(ssid).sheet-by-name("artists", true)
    end

select-columns(artists, [list: "c1", "c2"])

names = artists.get-column("c2")

elt1 = names.first
